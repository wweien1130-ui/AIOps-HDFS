#!/usr/bin/env python3
"""
实时异常预测服务
从 ClickHouse online.block_event_stats 增量读取 block 特征，
调用 MLP 模型预测异常概率，将结果写入 online.anomaly_blocks，
并更新 Redis 中的 Top N 异常缓存。
"""
```shell
import os
import time
import logging
import argparse
from datetime import datetime, timedelta

import redis
import clickhouse_connect
import pandas as pd
import joblib

# ==================== 默认配置 ====================
DEFAULT_CONFIG = {
    'clickhouse_host': '192.168.115.129',
    'clickhouse_port': 9000,
    'clickhouse_user': 'default',
    'clickhouse_password': '',
    'clickhouse_db': 'online',
    'redis_host': '192.168.115.129',
    'redis_port': 6379,
    'redis_db': 0,
    'redis_password': None,
    'redis_ttl_seconds': 3600,          # Hash 详情过期时间（秒）
    'model_path': '/path/to/block_anomaly_model.pkl',
    'scaler_path': '/path/to/scaler.pkl',
    'predict_interval': 30,              # 预测间隔（秒）
    'anomaly_threshold': 0.5,            # 异常概率阈值
    'top_n': 10,                         # Redis 中保留 Top N
    'last_processed_key': 'online:last_processed_time',  # Redis key 存储上次处理时间
    'redis_top_key': 'anomaly:top',      # Redis Sorted Set key
    'redis_detail_prefix': 'anomaly:detail:',  # Redis Hash 前缀
}

# ==================== 日志 ====================
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger('realtime_predictor')

# ==================== 服务类 ====================
class RealtimePredictor:
    def __init__(self, config):
        self.config = config
        self.ch_client = None
        self.redis_client = None
        self.model = None
        self.scaler = None
        self.last_processed_time = None

    def connect_clickhouse(self):
        """建立 ClickHouse 连接"""
        self.ch_client = clickhouse_connect.get_client(
            host=self.config['clickhouse_host'],
            port=self.config['clickhouse_port'],
            username=self.config['clickhouse_user'],
            password=self.config['clickhouse_password'],
            database=self.config['clickhouse_db']
        )
        logger.info("ClickHouse 连接成功")

    def connect_redis(self):
        """建立 Redis 连接"""
        try:
            self.redis_client = redis.Redis(
                host=self.config['redis_host'],
                port=self.config['redis_port'],
                db=self.config['redis_db'],
                password=self.config['redis_password'],
                decode_responses=True
            )
            self.redis_client.ping()
            logger.info("Redis 连接成功")
        except Exception as e:
            logger.error(f"Redis 连接失败: {e}")
            self.redis_client = None

    def load_model(self):
        """加载模型和标准化器"""
        self.model = joblib.load(self.config['model_path'])
        self.scaler = joblib.load(self.config['scaler_path'])
        logger.info("模型和标准化器加载完成")

    def get_last_processed_time(self):
        """从 Redis 获取上次处理的时间戳，若无则返回默认值（最近1小时）"""
        if self.redis_client:
            ts = self.redis_client.get(self.config['last_processed_key'])
            if ts:
                return ts
        # 默认处理最近1小时的数据，避免初次运行扫描全表
        default_ts = (datetime.now() - timedelta(hours=1)).strftime('%Y-%m-%d %H:%M:%S')
        logger.info(f"未找到上次处理时间，使用默认值: {default_ts}")
        return default_ts

    def save_last_processed_time(self, ts):
        """保存最后处理的时间戳到 Redis"""
        if self.redis_client:
            self.redis_client.set(self.config['last_processed_key'], ts)
            logger.debug(f"保存最后处理时间: {ts}")

    def fetch_incremental_blocks(self, last_ts):
        """
        查询自 last_ts 以来新增的 block 事件聚合（按 block_id 聚合 E1-E29）
        返回 DataFrame，包含 block_id, E1...E29, last_updated
        """
        # 注意：使用 last_updated > 时间戳，确保不重复
        query = f"""
        SELECT 
            block_id,
            sumIf(cnt, event_id='E1') AS E1,
            sumIf(cnt, event_id='E2') AS E2,
            sumIf(cnt, event_id='E3') AS E3,
            sumIf(cnt, event_id='E4') AS E4,
            sumIf(cnt, event_id='E5') AS E5,
            sumIf(cnt, event_id='E6') AS E6,
            sumIf(cnt, event_id='E7') AS E7,
            sumIf(cnt, event_id='E8') AS E8,
            sumIf(cnt, event_id='E9') AS E9,
            sumIf(cnt, event_id='E10') AS E10,
            sumIf(cnt, event_id='E11') AS E11,
            sumIf(cnt, event_id='E12') AS E12,
            sumIf(cnt, event_id='E13') AS E13,
            sumIf(cnt, event_id='E14') AS E14,
            sumIf(cnt, event_id='E15') AS E15,
            sumIf(cnt, event_id='E16') AS E16,
            sumIf(cnt, event_id='E17') AS E17,
            sumIf(cnt, event_id='E18') AS E18,
            sumIf(cnt, event_id='E19') AS E19,
            sumIf(cnt, event_id='E20') AS E20,
            sumIf(cnt, event_id='E21') AS E21,
            sumIf(cnt, event_id='E22') AS E22,
            sumIf(cnt, event_id='E23') AS E23,
            sumIf(cnt, event_id='E24') AS E24,
            sumIf(cnt, event_id='E25') AS E25,
            sumIf(cnt, event_id='E26') AS E26,
            sumIf(cnt, event_id='E27') AS E27,
            sumIf(cnt, event_id='E28') AS E28,
            sumIf(cnt, event_id='E29') AS E29,
            max(last_updated) AS last_updated
        FROM online.block_event_stats
        WHERE last_updated > '{last_ts}'
        GROUP BY block_id
        """
        df = self.ch_client.query_df(query)
        return df

    def predict_and_insert(self, df):
        """对 DataFrame 中的每个 block 预测异常，并将异常结果写入 online.anomaly_blocks"""
        if df.empty:
            return []
        feature_cols = [f'E{i}' for i in range(1, 30)]
        X = df[feature_cols].fillna(0).astype(float)
        X_scaled = self.scaler.transform(X)
        probs = self.model.predict_proba(X_scaled)[:, 1]   # 正类概率（异常）

        anomalies = []
        for idx, row in df.iterrows():
            score = probs[idx]
            if score > self.config['anomaly_threshold']:
                record = {
                    'block_id': row['block_id'],
                    **{f'E{i}': int(row.get(f'E{i}', 0)) for i in range(1, 30)},
                    'anomaly_score': score,
                    'detected_at': datetime.now().strftime('%Y-%m-%d %H:%M:%S.%f')[:-3]  # 毫秒精度
                }
                anomalies.append(record)
        if anomalies:
            # 批量插入 anomaly_blocks 表
            self.ch_client.insert_df('online.anomaly_blocks', pd.DataFrame(anomalies))
            logger.info(f"插入 {len(anomalies)} 条异常记录")
        return anomalies

    def update_redis_top_anomalies(self, anomalies):
        """更新 Redis 中的 Top N 异常 Sorted Set 和详情 Hash（带 TTL）"""
        if not self.redis_client or not anomalies:
            return
        pipe = self.redis_client.pipeline()
        for rec in anomalies:
            block_id = rec['block_id']
            score = rec['anomaly_score']
            # 更新 Sorted Set（分数覆盖）
            pipe.zadd(self.config['redis_top_key'], {block_id: score})
            # 存储详情 Hash，并设置 TTL
            detail = {f'E{i}': rec[f'E{i}'] for i in range(1, 30)}
            detail['anomaly_score'] = score
            hash_key = f"{self.config['redis_detail_prefix']}{block_id}"
            pipe.hset(hash_key, mapping=detail)
            pipe.expire(hash_key, self.config['redis_ttl_seconds'])
        # 保留 Top N，移除超出部分
        pipe.zremrangebyrank(self.config['redis_top_key'], 0, -(self.config['top_n']+1))
        # 可选：为 Sorted Set 也设置 TTL（避免无限增长，但通常不需要）
        # pipe.expire(self.config['redis_top_key'], self.config['redis_ttl_seconds'] * 2)
        pipe.execute()
        logger.info(f"Redis Top {self.config['top_n']} 异常更新完成")

    def run(self):
        """主循环"""
        logger.info("实时预测服务启动")
        self.connect_clickhouse()
        self.connect_redis()
        self.load_model()

        while True:
            try:
                last_ts = self.get_last_processed_time()
                logger.info(f"查询增量数据，上次处理时间: {last_ts}")
                df = self.fetch_incremental_blocks(last_ts)
                if not df.empty:
                    logger.info(f"获取到 {len(df)} 个 block 的新特征")
                    anomalies = self.predict_and_insert(df)
                    if anomalies:
                        self.update_redis_top_anomalies(anomalies)
                    # 更新最后处理时间（使用本次查询到的最大 last_updated）
                    new_last_ts = df['last_updated'].max()
                    self.save_last_processed_time(new_last_ts)
                    logger.info(f"更新最后处理时间为: {new_last_ts}")
                else:
                    logger.info("无新数据")
            except Exception as e:
                logger.exception(f"预测循环出错: {e}")
            time.sleep(self.config['predict_interval'])

# ==================== 主入口 ====================
def parse_args():
    parser = argparse.ArgumentParser(description='实时异常预测服务')
    parser.add_argument('--config', type=str, help='配置文件路径（JSON 或 YAML）')
    # 也可以直接通过命令行覆盖参数
    parser.add_argument('--model', default=DEFAULT_CONFIG['model_path'], help='模型文件路径')
    parser.add_argument('--scaler', default=DEFAULT_CONFIG['scaler_path'], help='标准化器路径')
    parser.add_argument('--interval', type=int, default=DEFAULT_CONFIG['predict_interval'], help='预测间隔（秒）')
    parser.add_argument('--threshold', type=float, default=DEFAULT_CONFIG['anomaly_threshold'], help='异常阈值')
    parser.add_argument('--top-n', type=int, default=DEFAULT_CONFIG['top_n'], help='Redis 保留 Top N')
    return parser.parse_args()

def load_config(args):
    config = DEFAULT_CONFIG.copy()
    # 如果提供了配置文件，可以加载（这里简化，直接使用命令行覆盖）
    if args.model:
        config['model_path'] = args.model
    if args.scaler:
        config['scaler_path'] = args.scaler
    config['predict_interval'] = args.interval
    config['anomaly_threshold'] = args.threshold
    config['top_n'] = args.top_n
    return config

if __name__ == '__main__':
    args = parse_args()
    config = load_config(args)
    predictor = RealtimePredictor(config)
    predictor.run()
```
